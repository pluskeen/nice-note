##### 二维数组行列转换
```js
/**
 * 二维数组行列转换
 * @param arr 二维数组，子数组长度相同
 */
rowColTrans(arr) {  
	return arr[0].map((c, i) => arr.map(row => row[i]))  
}
```

##### 数组转树
```js
/**
 * 数组转树
 * @param list 包含类似id、pid父子关系的数组
 */
listToTree(list) {  
	let map = {};
	let node;
	let tree = [];

	for (let i = 0; i < list.length; i++) {
		map[list[i].id] = list[i];
		list[i].children = [];
	}

	for (let i = 0; i < list.length; i += 1) {
		node = list[i];
		// 根节点id -> -1
		if (node.pid !== -1) {
			map[node.pid].children.push(node);
		} else {
			tree.push(node);
		}
	} 
	return tree;  
}

/**
 * 数组转树带层级
 * @param list 包含类似id、pid父子关系的数组
 * @param pid 父id
 * @param level 层级，初始值 0
 */
listToTreeWithLevel(list, pid, level = 0) {  
	let out = [];
	for (const node of list) {
		if (node.pid === pid) {
			node.level = level;
			const children = listToTreeWithLevel(list, node.id, level + 1);
			if (children.length) {
				node.children = children;
			}
			out.push(node);
		}
	} 
	return out  
}
```

##### 树转数组
```js
/**
 * 树转数组
 * @param tree 树
 */
treeToList(tree) {  
	const queen = [];
	const out = [];
	queen = queen.concat(tree); 
	while(queen.length) {
		const first = queen.shift();
		if (first.children) {
			queen = queen.concat(first.children);
			delete first['children'];
		}
		out.push(first);
	}
	return out;  
}
```

##### 判断字符串是否为合法日期格式
```js
/**
 * 判断字符串是否为合法日期格式
 * @param str 字符串
 * @return boolean true -> 合法日期格式
 */
judgeStrIsDateFormat(str) {
	return !isNaN(Date.parse(str));
}
```

##### 使用Boolean构造函数判断假值
```js
/**  
 * 判断假值  
 * @param val 要判断的值  
 */
judgeFalseValue(val) {
	return Boolean(val);  
}
```

##### 数字十位补零
```js
/**  
 * 数字十位补零  
 * @param num 数字  
 * @param length 补零后的位数  
 */  
patchZero(num, length = 2) {
	if (!!String.prototype.padStart) {
		return (`${num}`).padStart(length, '0');
	}
	return (`0${num}`).slice(-length);
}
```
